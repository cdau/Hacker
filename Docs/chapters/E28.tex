\section{Aufgabe E28}
Die Verteilung des Zufallsvektors $(X,Y)$ sei gegeben durch
\begin{onehalfspace}
\begin{center}
$P(X = -1, Y = 1) = \frac{1}{8} = P(X = 0, Y = 1)$

$P(X = 1, Y = -1) = \frac{1}{8} = P(X = 0, Y = -1)$

$P(X = 2, Y = 0) = \frac{1}{4} = P(X = -1, Y = 0)$.
\end{center}
\end{onehalfspace}

\noindent Bestimmen Sie
\begin{enumerate} [leftmargin=1cm, label=\alph*)]
	\item Die (Rand-)Verteilung von $X$
	\vspace{4cm}
	\item $E(X)$ und $Var(X)$
	\vspace{4cm}
	\item $E(X \cdot Y)$ und $Var(X \cdot Y)$
	\vspace{4cm}
	\item $E(6 \cdot X^2 - 8X + 5Y - 4 \cdot XY)$
\end{enumerate}
